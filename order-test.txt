// 2. Facebook Login & Embedded Signup Logic    
function launchFBLogin(type)
{
    let scopes = '';

    if (type === 'messenger')
    {
        scopes = 'pages_show_list,pages_messaging,pages_read_engagement,pages_manage_metadata';
    }
    else if (type === 'comment')
    {
        scopes = 'pages_show_list,pages_read_engagement,pages_manage_engagement';
    }
    else if (type === 'whatsapp')
    {
        scopes = 'whatsapp_business_management,whatsapp_business_messaging';
    }

    FB.login(function(response)
    {
        if (response.authResponse)
        {
            const userToken = response.authResponse.accessToken;

            document.getElementById('fb_short_token').value = userToken;

            if (type === 'messenger' || type === 'comment')
                fetchPages(userToken);

            else if (type === 'whatsapp')
                fetchWABA(userToken);
        }

    }, { scope: scopes });
}
    
// 3. Fetch WhatsApp Details (Embedded Signup)    
function fetchWABA(token) {    
    FB.api('/me/whatsapp_business_accounts', { access_token: token }, function(res) {    
        if (res.data && res.data.length > 0) {    
            const wabaId = res.data[0].id;    
            document.getElementById('wa_waba_id').value = wabaId;    
            document.getElementById('wa_access_token').value = token; // Sementara pake short token    
                
            // Get Phone Numbers    
            FB.api(`/${wabaId}/phone_numbers`, { access_token: token }, function(pnRes) {    
                if (pnRes.data && pnRes.data.length > 0) {    
                    document.getElementById('wa_phone_number_id').value = pnRes.data[0].id;    
                    alert('WhatsApp Business Connected!');    
                }    
            });    
        }    
    });    
}    
    
// 4. Fetch Messenger Pages    
function fetchPages(userToken)
{
    FB.api('/me/accounts',
    { access_token: userToken },
    function(res)
    {
        if (!res.data) return;

        const select = document.getElementById('pageSelect');
        const div = document.getElementById('fb-page-list');

        select.innerHTML='';

        res.data.forEach(page =>
        {
            const opt = new Option(page.name, page.id);

            opt.dataset.token = page.access_token;
            opt.dataset.name  = page.name;

            select.add(opt);
        });

        div.style.display='block';

        select.onchange = function()
        {
            const opt = select.options[select.selectedIndex];

            document.getElementById('page_access_token').value =
                opt.dataset.token;

            document.getElementById('page_name').value =
                opt.dataset.name;
        };

        select.onchange();
    });
}

<?php

class ChatbotChannelHandler
{
    private const FB_APP_ID     = 'YOUR_APP_ID';
    private const FB_APP_SECRET = 'YOUR_APP_SECRET';


    public static function handle(int $user_id, array $product): never
    {
        DB::begin();

        try {

            $init     = ClientService::initialStatus($product);

            $service  = $product['service'];

            $provider = $_POST['provider'] ?? 'meta';


            $credentials = match($service)
            {

                'messenger' => self::messengerCredentials(),

                'whatsapp'  => self::whatsappCredentials(),

                'telegram'  => self::telegramCredentials(),

                default => []
            };


            $meta = $init['meta'];

            $meta['ai_engine'] =
            [
                'provider' => $_POST['ai_provider'] ?? null,
                'model'    => $_POST['ai_model'] ?? null
            ];


            DB::execute("
                INSERT INTO clients
                (
                    user_id,
                    product_id,
                    name,
                    service,
                    provider,
                    credentials,
                    api_key,
                    status,
                    expired_at,
                    meta
                )
                VALUES (?,?,?,?,?,?,?,?,?,?)
            ",
            [
                $user_id,
                $product['id'],
                $_POST['name'],

                $service,
                $provider,

                json_encode($credentials),

                ClientService::apiKey(),

                $init['status'],
                $init['expired_at'],

                json_encode($meta)
            ]);


            $id = DB::lastId();

            DB::commit();


            redirect("client-detail.php?id={$id}");

            exit;

        }
        catch(Throwable $e)
        {
            DB::rollback();

            die($e->getMessage());
        }
    }



    private static function messengerCredentials(): array
    {
        $page_id   = $_POST['page_id'] ?? '';
        $page_name = $_POST['page_name'] ?? '';

        $page_token = $_POST['page_access_token'] ?? '';


        if (!$page_id || !$page_token)
            throw new Exception("Facebook Page belum dipilih");


        $longToken =
            self::exchangeFBToken($page_token)
            ?? $page_token;


        return
        [
            'page_id'      => $page_id,

            'page_name'    => $page_name,

            'access_token' => $longToken
        ];
    }



    private static function whatsappCredentials(): array
    {
        return
        [
            'phone_number_id'
                => $_POST['phone_number_id'] ?? '',

            'waba_id'
                => $_POST['waba_id'] ?? '',

            'access_token'
                => $_POST['access_token'] ?? ''
        ];
    }



    private static function telegramCredentials(): array
    {
        return
        [
            'bot_token'
                => $_POST['bot_token'] ?? ''
        ];
    }



    private static function exchangeFBToken(string $token): ?string
    {
        $url =
        "https://graph.facebook.com/v18.0/oauth/access_token?"
        . http_build_query(
        [
            'grant_type'        => 'fb_exchange_token',

            'client_id'        => self::FB_APP_ID,

            'client_secret'    => self::FB_APP_SECRET,

            'fb_exchange_token'=> $token
        ]);


        $res = file_get_contents($url);

        if (!$res) return null;


        $json = json_decode($res,true);

        return $json['access_token'] ?? null;
    }

}
